## Banco de Dados 
-- 1. Tabela de Imóveis (Properties)
create table properties (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  title text not null, -- Título do anúncio
  description text,
  price numeric not null, -- Valor do aluguel/venda
  condo_price numeric default 0, -- Condomínio
  iptu_price numeric default 0, -- IPTU
  address text, -- Rua Senador Pompeu...
  neighborhood text, -- Benfica
  city text default 'Fortaleza',
  state text default 'CE',
  area integer, -- Metragem
  bedrooms integer, -- Quartos
  bathrooms integer, -- Banheiros
  parking integer, -- Vagas
  images text[], -- Array de URLs das fotos
  type text default 'apartamento', -- casa, apartamento, comercial
  featured boolean default false -- Destaque na home
);

-- 2. Tabela de Propostas (CRM)
create table proposals (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  property_id bigint references properties(id) on delete cascade, -- Se apagar imóvel, apaga propostas
  client_name text not null,
  client_email text,
  client_phone text,
  entry_offer numeric, -- Valor de Entrada
  total_offer numeric, -- Valor Proposto
  status text default 'novo', -- novo, negociacao, aceito, recusado
  notes text -- Anotações do corretor
);

-- 3. Habilitar Realtime (Para o Dashboard atualizar sozinho)
alter publication supabase_realtime add table proposals;

-- 4. Criar Bucket para Imagens
insert into storage.buckets (id, name, public) values ('imovit-images', 'imovit-images', true);

-- 5. Política de segurança (Permitir uploads públicos por enquanto - MVP)
create policy "Public Access"
  on storage.objects for select
  using ( bucket_id = 'imovit-images' );

create policy "Public Upload"
  on storage.objects for insert
  with check ( bucket_id = 'imovit-images' );
